from education_platform_client import ApiClient, Configuration
from education_platform_client.api import courses_api

# Налаштування клієнта

    access_token="YOUR_JWT_TOKEN"
)

client = ApiClient(config)
courses = courses_api.CoursesApi(client)

# Отримання списку курсів
response = courses.get_courses()
